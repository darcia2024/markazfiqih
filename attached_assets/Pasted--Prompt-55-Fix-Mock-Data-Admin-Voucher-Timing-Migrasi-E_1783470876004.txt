```ts
export async function getAdminDashboardSummary() {
  const [classesRes, publishedRes, pendingInvoicesRes, paidInvoicesRes, usersRes] = await Promise.all([
    supabase.from('classes').select('id', { count: 'exact', head: true }),
    supabase.from('classes').select('id', { count: 'exact', head: true }).eq('status', 'published'),
    supabase.from('invoices').select('id', { count: 'exact', head: true }).eq('status', 'pending'),
    supabase.from('invoices').select('id', { count: 'exact', head: true }).eq('status', 'paid'),
    supabase.from('user_profiles').select('user_id', { count: 'exact', head: true }),
  ]);

  return {
    totalClasses: classesRes.count ?? 0,
    publishedClasses: publishedRes.count ?? 0,
    draftClasses: (classesRes.count ?? 0) - (publishedRes.count ?? 0),
    pendingOrders: pendingInvoicesRes.count ?? 0,
    totalPaidOrders: paidInvoicesRes.count ?? 0,
    totalUsers: usersRes.count ?? 0,
  };
}
```

### 2a. Tambah fungsi di `src/lib/db.ts`
```ts
export async function listAllInvoicesForAdmin() {
  const { data, error } = await supabase
    .from('invoices')
    .select(`
      id, user_id, total_amount, status, mayar_invoice_id, created_at, paid_at,
      invoice_items ( id, class_id, bundle_id, price, classes ( title ), bundles ( title ) )
    `)
    .order('created_at', { ascending: false });
  if (error) throw error;

  return (data ?? []).map((inv: any) => ({
    id: inv.id as string,
    userId: inv.user_id as string,
    totalAmount: inv.total_amount as number,
    status: inv.status as 'pending' | 'paid' | 'failed',
    mayarInvoiceId: inv.mayar_invoice_id as string | null,
    createdAt: inv.created_at as string,
    paidAt: inv.paid_at as string | null,
    items: (inv.invoice_items ?? []).map((item: any) => ({
      id: item.id as string,
      title: item.classes?.title ?? item.bundles?.title ?? '(tidak diketahui)',
      price: item.price as number,
    })),
  }));
}
```

## Section 3 — Pindahkan Increment `used_count` Voucher ke `simulate-success`

Voucher sekarang di-burn (increment `used_count`) saat **checkout** (invoice
dibuat), bukan saat **pembayaran sukses**. Kalau user tidak menyelesaikan
pembayaran, kuota voucher tetap terpakai sia-sia.

### 3a. Migrasi kolom (jalankan manual, SQL disiapkan terpisah)
Perlu kolom baru untuk menyimpan referensi voucher yang dipakai pada
sebuah invoice, supaya `simulate-success` tahu voucher mana yang harus
di-increment. Siapkan SQL berikut untuk saya jalankan manual:
```sql
ALTER TABLE invoices ADD COLUMN IF NOT EXISTS voucher_id UUID REFERENCES class_vouchers(id) ON DELETE SET NULL;
```

### 3b. Edit `supabase/functions/checkout/index.ts`
- Tetap validasi voucher dan hitung harga seperti sekarang.
- HAPUS pemanggilan `.rpc('increment_voucher_usage', ...)` dari sini —
  jangan increment saat checkout lagi.
- Saat insert row ke tabel `invoices`, sertakan `voucher_id: voucherRow.id`
  kalau voucher dipakai (null kalau tidak).

### 3c. Edit `supabase/functions/simulate-success/index.ts`
- Setelah update invoice jadi `paid` dan sebelum/sesudah membuat
  enrollment, cek apakah invoice ini punya `voucher_id`. Kalau ada,
  panggil `.rpc('increment_voucher_usage', { voucher_id: invoice.voucher_id })`
  di sini.

**PENTING**: setelah edit dua edge function ini, SAYA (Darcia) perlu
deploy ulang keduanya secara manual lewat Supabase Dashboard atau CLI —
edge function tidak auto-deploy dari perubahan kode di repo. Tandai ini
dengan jelas di ringkasan akhir.

## Section 5 — Bersihkan Policy Duplikat `user_profiles`

Jalankan dulu untuk melihat semua policy:

hapus yang duplikat, sisakan satu. Untuk UPDATE, PASTIKAN yang disisakan

dilindungi trigger `prevent_self_admin_escalation` dari perbaikan
